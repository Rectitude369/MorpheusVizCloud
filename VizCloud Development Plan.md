Executive Summary                                                                                                                                                                                       
                                                                                                                                                                                                        
I've analyzed the existing morpheus-hvm-manager.sh (2,744 lines) and DEVELOPMENT.md. This is a robust terminal-based tool managing Morpheus HVM infrastructure. VizCloud will transform this into a     
production-grade Electron application with a beautiful dark-themed GUI, real-time dashboards, animated network topology maps, and enterprise-grade reliability.

1.3 Database Schema (SQLite)                                                                                                                                                                            
                                                                                                                                                                                                        
```sql                                                                                                                                                                                                  
  -- HOSTS TABLE                                                                                                                                                                                        
  CREATE TABLE hosts (                                                                                                                                                                                  
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      hostname TEXT UNIQUE NOT NULL,                                                                                                                                                                    
      ip_address TEXT NOT NULL,                                                                                                                                                                         
      datacenter TEXT,                                                                                                                                                                                  
      status TEXT DEFAULT 'offline', -- online, offline, degraded, maintenance                                                                                                                          
      last_seen DATETIME,                                                                                                                                                                               
      cpu_model TEXT,                                                                                                                                                                                   
      cpu_cores INTEGER,                                                                                                                                                                                
      total_memory BIGINT,                                                                                                                                                                              
      total_storage BIGINT,                                                                                                                                                                             
      os_version TEXT,                                                                                                                                                                                  
      morpheus_version TEXT,                                                                                                                                                                            
      ssh_key_id INTEGER,                                                                                                                                                                               
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP,                                                                                                                                                    
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                     
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- CLUSTERS TABLE                                                                                                                                                                                     
  CREATE TABLE clusters (                                                                                                                                                                               
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      name TEXT UNIQUE NOT NULL,                                                                                                                                                                        
      datacenter TEXT,                                                                                                                                                                                  
      quorum_status TEXT, -- quorate, not-quorate                                                                                                                                                       
      total_nodes INTEGER,                                                                                                                                                                              
      online_nodes INTEGER,                                                                                                                                                                             
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP,                                                                                                                                                    
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                     
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- CLUSTER_HOSTS (Junction Table)                                                                                                                                                                     
  CREATE TABLE cluster_hosts (                                                                                                                                                                          
      cluster_id INTEGER NOT NULL,                                                                                                                                                                      
      host_id INTEGER NOT NULL,                                                                                                                                                                         
      role TEXT, -- primary, secondary, standby                                                                                                                                                         
      FOREIGN KEY (cluster_id) REFERENCES clusters(id),                                                                                                                                                 
      FOREIGN KEY (host_id) REFERENCES hosts(id),                                                                                                                                                       
      PRIMARY KEY (cluster_id, host_id)                                                                                                                                                                 
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- VIRTUAL MACHINES TABLE                                                                                                                                                                             
  CREATE TABLE vms (                                                                                                                                                                                    
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      host_id INTEGER NOT NULL,                                                                                                                                                                         
      name TEXT NOT NULL,                                                                                                                                                                               
      uuid TEXT UNIQUE,                                                                                                                                                                                 
      status TEXT DEFAULT 'shut off', -- running, shut off, paused, crashed                                                                                                                             
      vcpus_current INTEGER,                                                                                                                                                                            
      vcpus_max INTEGER,                                                                                                                                                                                
      memory BIGINT,                                                                                                                                                                                    
      autostart BOOLEAN DEFAULT 0,                                                                                                                                                                      
      description TEXT,                                                                                                                                                                                 
      FOREIGN KEY (host_id) REFERENCES hosts(id),                                                                                                                                                       
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP,                                                                                                                                                    
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                     
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- VM DISKS TABLE                                                                                                                                                                                     
  CREATE TABLE vm_disks (                                                                                                                                                                               
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      vm_id INTEGER NOT NULL,                                                                                                                                                                           
      device TEXT NOT NULL, -- vda, vdb, etc.                                                                                                                                                           
      source TEXT NOT NULL,                                                                                                                                                                             
      size BIGINT,                                                                                                                                                                                      
      format TEXT,                                                                                                                                                                                      
      FOREIGN KEY (vm_id) REFERENCES vms(id)                                                                                                                                                            
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- VM NETWORKS TABLE                                                                                                                                                                                  
  CREATE TABLE vm_networks (                                                                                                                                                                            
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      vm_id INTEGER NOT NULL,                                                                                                                                                                           
      interface TEXT NOT NULL, -- vnet0, eth0, etc.                                                                                                                                                     
      mac_address TEXT,                                                                                                                                                                                 
      ip_address TEXT,                                                                                                                                                                                  
      network_name TEXT,                                                                                                                                                                                
      FOREIGN KEY (vm_id) REFERENCES vms(id)                                                                                                                                                            
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- METRICS HISTORY TABLE                                                                                                                                                                              
  CREATE TABLE metrics_history (                                                                                                                                                                        
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      host_id INTEGER NOT NULL,                                                                                                                                                                         
      timestamp DATETIME NOT NULL,                                                                                                                                                                      
      cpu_usage REAL,                                                                                                                                                                                   
      memory_used BIGINT,                                                                                                                                                                               
      memory_total BIGINT,                                                                                                                                                                              
      storage_used BIGINT,                                                                                                                                                                              
      storage_total BIGINT,                                                                                                                                                                             
      vm_count INTEGER,                                                                                                                                                                                 
      running_vm_count INTEGER,                                                                                                                                                                         
      FOREIGN KEY (host_id) REFERENCES hosts(id)                                                                                                                                                        
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- MIGRATION HISTORY TABLE                                                                                                                                                                            
  CREATE TABLE migration_history (                                                                                                                                                                      
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      vm_name TEXT NOT NULL,                                                                                                                                                                            
      source_host TEXT NOT NULL,                                                                                                                                                                        
      target_host TEXT NOT NULL,                                                                                                                                                                        
      status TEXT NOT NULL, -- success, failed                                                                                                                                                          
      duration TEXT,                                                                                                                                                                                    
      data_transferred BIGINT,                                                                                                                                                                          
      timestamp DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                      
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- OPERATIONS LOG TABLE                                                                                                                                                                               
  CREATE TABLE operations_log (                                                                                                                                                                         
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      operation_type TEXT NOT NULL,                                                                                                                                                                     
      target_type TEXT, -- host, vm, cluster                                                                                                                                                            
      target_id INTEGER,                                                                                                                                                                                
      status TEXT NOT NULL, -- success, failed, pending                                                                                                                                                 
      details TEXT,                                                                                                                                                                                     
      error_message TEXT,                                                                                                                                                                               
      timestamp DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                      
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- SSH KEYS TABLE                                                                                                                                                                                     
  CREATE TABLE ssh_keys (                                                                                                                                                                               
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      host_id INTEGER UNIQUE,                                                                                                                                                                           
      private_key TEXT NOT NULL, -- Encrypted                                                                                                                                                           
      public_key TEXT NOT NULL,                                                                                                                                                                         
      passphrase_hint TEXT,                                                                                                                                                                             
      created_at DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                     
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- USER SETTINGS TABLE                                                                                                                                                                                
  CREATE TABLE user_settings (                                                                                                                                                                          
      key TEXT PRIMARY KEY,                                                                                                                                                                             
      value TEXT,                                                                                                                                                                                       
      updated_at DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                     
  );                                                                                                                                                                                                    
                                                                                                                                                                                                        
  -- NOTIFICATIONS TABLE                                                                                                                                                                                
  CREATE TABLE notifications (                                                                                                                                                                          
      id INTEGER PRIMARY KEY AUTOINCREMENT,                                                                                                                                                             
      type TEXT NOT NULL, -- info, warning, error, success                                                                                                                                              
      title TEXT NOT NULL,                                                                                                                                                                              
      message TEXT NOT NULL,                                                                                                                                                                            
      read BOOLEAN DEFAULT 0,                                                                                                                                                                           
      action_required BOOLEAN DEFAULT 0,                                                                                                                                                                
      timestamp DATETIME DEFAULT CURRENT_TIMESTAMP                                                                                                                                                      
  );                                                                                                                                                                                                    
```
